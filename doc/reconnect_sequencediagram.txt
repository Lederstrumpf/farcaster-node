t_wallet -> t_farcasterd : Ctl Keys
t_farcasterd <- t_farcasterd : Ctl ReconnectPeer (continues)
t_farcasterd -> t_farcasterd : launch\npeerd connect
==Taker peerd restarted, connects to maker forked peerd==
t_peerd -> m_peerd : NodeId (raw)
m_peerd -> m_farcasterd: Ctl Hello
m_farcasterd -> m_swapd: Ctl PeerdReconnected
